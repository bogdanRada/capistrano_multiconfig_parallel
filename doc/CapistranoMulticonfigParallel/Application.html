def invoke_task(task_string)
  tasks = top_level_tasks
  if multi_interactive?(tasks)
    application = tasks.first.split(':').reverse[1]
    stage = tasks.first.split(':').reverse[0]
    name, args = parse_task_string(tasks.second)
    CapistranoMulticonfigParallel.create_debug_log
    CapistranoMulticonfigParallel.set_celluloid_logger if CapistranoMulticonfigParallel::CelluloidManager.debug_enabled
    CapistranoMulticonfigParallel::JobManager.new(application, stage, name, args, handle_options)
  else
    super(task_string)
  end
end

def multi_debug
  ['--multi-debug', '-D',
   'Sets the debug enabled for celluloid actors',
   lambda do |_value|
     CapistranoMulticonfigParallel::CelluloidManager.debug_enabled = true
     Celluloid.task_class = Celluloid::TaskThread
   end
  ]
end

def multi_interactive?(tasks)
  tasks.size > 1 && stages.include?(tasks.first) && CapistranoMulticonfigParallel.show_celulloid_menu? && ENV[CapistranoMulticonfigParallel.env_key_job_id].blank?
end
