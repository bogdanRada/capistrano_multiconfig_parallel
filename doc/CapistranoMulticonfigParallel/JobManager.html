def initialize(application, stage, name, args, argv)
  @manager = CapistranoMulticonfigParallel::CelluloidManager.new
  @deps = CapistranoMulticonfigParallel::MenuManager.new
  @application = application
  @stage = stage
  @name = name
  @args = args
  @argv = multi_fetch_argv(argv).except('BOX')
  CapistranoMulticonfigParallel.enable_skipping_menu
  fetch_dependencies
  deploy_current_app
end

def call_task_deploy_app(options = {})
  main_box_name = ENV['BOX'].blank? ? '' : ENV['BOX']
  if @stage == 'development' && /^[a-z0-9,]+/.match(main_box_name)
    execute_on_multiple_boxes(main_box_name, options)
  elsif @stage != 'development' # no multiple boxes, just one
    execute_celluloid_task(options.merge(final_job: options.fetch(:final_job, false)))
  else
    raise "Incorrect box name #{main_box_name}. Should contain only small letters and digits"
  end
end

def deploy_current_app
  return if CapistranoMulticonfigParallel.interactive_menu?
  ENV['BRANCH'] = ENV[CapistranoMulticonfigParallel.env_key_branch_backup] if ENV[CapistranoMulticonfigParallel.env_key_branch_backup].present?
  branch = { @application => ENV['BRANCH'].present? ? ENV['BRANCH'] : '' }
  call_task_deploy_app(
    branches: branch,
    app: { app: @application },
    final_job: true,
    task_arguments: @args,
    env_options: @argv
  )
  sleep
end

def execute_celluloid_task(options)
  application_branches = options.fetch(:branches, {})
  app = options.fetch(:app, '')
  branch_name = application_branches[app[:app]].blank? ? '' : application_branches[app[:app]]
  env_opts = get_app_additional_env_options(app, @stage, options[:env_options]['BOX'])
  options[:env_options] = options[:env_options].reverse_merge(env_opts)
  if CapistranoMulticonfigParallel.execute_in_sequence
    execute_in_sequence(app, branch_name, options)
  else
    execute_with_celluloid(app, branch_name, options)
  end
end

def execute_in_sequence(app, branch_name, options)
  execute_standard_deploy(app: app,
                          stage: @stage,
                          env_options: {
                            'BRANCH' => branch_name,
                            'BOX' => ENV['BOX']
                          }.merge(options.fetch(:env_options, {})),
                          task_arguments: options.fetch(:task_arguments, []))
end

def execute_with_celluloid(app, branch_name, options)
  job = {
    app: app[:app],
    env: @stage,
    action: @name,
    task_arguments: options.fetch(:task_arguments, []),
    env_options: {
      'BRANCH' => branch_name,
      CapistranoMulticonfigParallel.env_key_skip_menu => true
    }.merge(options.fetch(:env_options, {})),
    final_job: options.fetch(:final_job, false)
  }
  @manager.delegate(job)
end

def fetch_dependencies
  applications = @deps.fetch_apps_needed_for_deployment(@application, @name)
  return unless applications.present?
  if !CapistranoMulticonfigParallel.interactive_menu? && ENV['BRANCH'].present?
    ENV[CapistranoMulticonfigParallel.env_key_branch_backup] = ENV['BRANCH']
    ENV['BRANCH'] = nil
  end
  application_branches = get_applications_branches(applications)
  applications.each do |app|
    call_task_deploy_app(
      branches: application_branches,
      app: app,
      task_arguments: @args,
      env_options: @argv
    )
  end
end
