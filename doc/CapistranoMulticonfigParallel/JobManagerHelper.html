def cap_log_message(message)
  error_message = message.respond_to?(:message) ? message.message : message.inspect
  err_backtrace = message.respond_to?(:backtrace) ? message.backtrace.join("\n\n") : ''
  message = "A #{message.class} occured: #{error_message} \n\n\n with backtrace: \n\n\  #{err_backtrace}"
  CapistranoMulticonfigParallel.logger.debug(message)
end

def execute_on_multiple_boxes(main_box_name, options)
  boxes = parse_inputted_value(value: main_box_name).split(',').compact
  boxes.each_with_index do |box_name, index|
    final_app_job = options.fetch(:final_job, false)
    final_job = (final_app_job.to_s == 'true' && index == (boxes.size - 1))
    options[:env_options]['BOX'] = box_name
    execute_celluloid_task(options.merge(final_job: final_job))
  end
end

def execute_standard_deploy(options)
  app = options.fetch(:app, {})[:app]
  stage = options.fetch(:stage, 'deploy')
  action = "#{@name}[#{options.fetch(:task_arguments, []).join(',')}]"
  arguments = setup_command_line_standard(options.fetch(:env_options, {}))

  command = "bundle exec cap #{app}:#{stage} #{action}  #{arguments}"
  puts("\n\n\n Executing #{command} \n\n\n .")
  sh("#{command}")
rescue => ex
  cap_log_message(ex)
  if @name == 'deploy'
    begin
      action = "deploy:rollback[#{options.fetch(:task_arguments, []).join(',')}]"
      command = "bundle exec cap #{app}:#{stage} #{action} #{arguments}"
      puts("\n\n\n Executing #{command} \n\n\n .")
      sh("#{command}")
    rescue => exception
      cap_log_message(exception)
      # nothing to do if rollback fails
    end
  end
end

def get_app_additional_env_options(app, stage, box = nil)
  options = {}
  if box.present?
    set :app_additional_env_options, ask_confirm("Please write additional ENV options for #{app[:app].camelcase} for BOX #{box}:", nil)
  else
    set :app_additional_env_options, ask_confirm("Please write additional ENV options for #{app[:app].camelcase} on stage #{stage}:", nil)
  end
  if fetch(:app_additional_env_options).present?
    env_options = parse_inputted_value(key: :app_additional_env_options)
    env_options = env_options.split(' ')
    options = multi_fetch_argv(env_options)
  end
  options
end

def get_applications_branches(apps)
  application_keys = {}
  apps.each do |app|
    if ENV['BRANCH'].blank? || (ENV['BRANCH'].present? && !CapistranoMulticonfigParallel.interactive_menu?)
      set :app_branch_name, ask("Branch name for #{app[:app].camelcase}:", nil)
      application_keys[app[:app]] = parse_inputted_value
    else
      application_keys[app[:app]] = ENV['BRANCH']
    end
  end
  application_keys
end

def multi_fetch_argv(args)
  options = {}
  args.each do |arg|
    if arg =~ /^(\w+)=(.*)$/m
      options[Regexp.last_match(1)] = Regexp.last_match(2)
    end
  end
  options
end

def parse_inputted_value(options = {})
  options = options.stringify_keys
  value = options['value'].present? ? options['value'] : fetch(options.fetch('key', :app_branch_name))
  if value.present?
    branch = value.gsub("\r\n", '')
    branch = branch.gsub("\n", '') if branch.present?
    branch = branch.gsub(/\s+/, ' ') if branch.present?
    branch = branch.strip if branch.present?
    return branch
  else
    return ''
  end
end

def setup_command_line_standard(options)
  opts = ''
  options.each do |key, value|
    opts << "#{key}=#{value} "
  end
  opts
end
