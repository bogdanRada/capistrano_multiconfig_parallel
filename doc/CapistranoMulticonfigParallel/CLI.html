def self.start
  if $stdin.isatty
    $stdin.sync = true
  end
  if $stdout.isatty
    $stdout.sync = true
  end
  unless ARGV.nil?
    ARGV.delete_if do |arg|
      CapistranoMulticonfigParallel.excluded_argv.map(&:downcase).include?(arg.downcase)
    end
  end
  CapistranoMulticonfigParallel.configuration_valid?
  CapistranoMulticonfigParallel.verify_app_dependencies(stages) if CapistranoMulticonfigParallel.configuration.track_dependencies
  CapistranoMulticonfigParallel::Application.new.run
rescue Interrupt
  `stty icanon echo`
  $stderr.puts 'Command cancelled.'
rescue => error
  $stderr.puts error
  exit(1)
end
